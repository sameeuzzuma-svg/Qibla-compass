// Qibla location (Kaaba in Makkah) const QIBLA_LAT = 21.4225; const QIBLA_LON = 39.8262; let userLat = null; let userLon = null; // Get user location if (navigator.geolocation) { navigator.geolocation.getCurrentPosition(pos => { userLat = pos.coords.latitude; userLon = pos.coords.longitude; }, err => { document.getElementById("info").innerText = "Location access denied!"; }); } // Calculate bearing between two coordinates function calculateQiblaDirection(lat1, lon1, lat2, lon2) { const toRad = deg => deg * Math.PI / 180; const toDeg = rad => rad * 180 / Math.PI; const dLon = toRad(lon2 - lon1); const y = Math.sin(dLon) * Math.cos(toRad(lat2)); const x = Math.cos(toRad(lat1)) * Math.sin(toRad(lat2)) - Math.sin(toRad(lat1)) * Math.cos(toRad(lat2)) * Math.cos(dLon); let brng = toDeg(Math.atan2(y, x)); return (brng + 360) % 360; } // Compass rotation window.addEventListener("deviceorientation", function(event) { if (userLat && userLon) { let heading = event.alpha; // device orientation if (typeof event.webkitCompassHeading !== "undefined") { heading = event.webkitCompassHeading; // iOS } let qiblaAngle = calculateQiblaDirection(userLat, userLon, QIBLA_LAT, QIBLA_LON); let rotation = qiblaAngle - heading; document.getElementById("needle").style.transform = `translate(-50%, -100%) rotate(${rotation}deg)`; document.getElementById("info").innerText = "Qibla direction: " + Math.round(qiblaAngle) + "° from North"; } }, true);
